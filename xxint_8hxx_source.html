136 template<class A, class B>
137 constexpr bool is_as_wide_as()
138 {
139  if (std::is_signed<B>::value == std::is_signed<A>::value)
140  return sizeof(B) <= sizeof(A);
141 
142  if (std::is_unsigned<B>::value && std::is_signed<A>::value)
143  return sizeof(B) < sizeof(A);
144 
145  return false;
146 }

151 {
152  if (std::is_unsigned<A>::value) return false;
