8 /// \mainpage int++: checked integers for C++
9 ///
10 /// ## Get it
11 ///
12 /// The latest version is available [on GitHub](https://github.com/tov/intpp).
13 ///
14 /// ## Use it
15 ///
16 /// The main class template is `intpp::Checked`. For any built-in integer-type
17 /// `T`, an `intpp::Checked<T>` behaves like `T` but it throws an exception
18 /// when the result of an operation would not fit in type `T`.
19 ///
20 /// For example, here is a function that computes the factorial of an `int`,
21 /// throwing an `intpp::overflow_too_large` exception if type `int` is not large
22 /// enough to hold the result:
23 ///
24 /// ```cpp
25 /// #include <int++.h>
26 ///
27 /// using namespace intpp;
28 ///
29 /// int factorial(int n)
30 /// {
31 /// Checked<int> result = 1;
32 ///
33 /// for (int i = 1; i <= n; ++i)
34 /// result *= i;
35 /// }
36 ///
37 /// return result.get()
38 /// }
39 /// ```
40 ///
41 /// If we want `factorial` to saturate and return `INT_MAX` instead of
42 /// overflowing, we can replace `Checked<int>` with
43 /// `Checked<int, policy::saturating>`. (There's a type alias, so we can
44 /// also write `Saturating<int>` instead for this type.)
45 ///
46 /// The library also provides checked conversions and mathematically-correct
47 /// mixed-sign comparisons between `Checked` types.
48 /// See the [`intpp`](namespaceintpp.html) namespace, which contains
49 /// everything this library defines.
50

87 /// Policies are class templates that specify how the `Checked` class
88 /// template behaves.
89 ///
90 /// A policy is parametrized by the underlying checked type. It then specifies:
91 ///
92 /// - Whether the type should wrap.
93 ///
94 /// - What the type should do or return on division by zero.
95 ///
96 /// - If non-wrapping, what the type should do or return when an operation
97 /// returns a value too small or too large for the type.
98 ///
99 /// A policy is used by supplying it as the second argument to `Checked`. For
100 /// example, we can make a saturating short by writing
101 ///
102 /// ```cpp
103 /// Checked<short, policy::saturating> cs;
104 /// ```
105 namespace policy {
106 
107 /// Saturates on overflow, throws on divide-by-zero.
108 template<class T>
109 struct saturating
110 {
111  /// Indicates that this policy does not wrap around.
112  static constexpr bool is_wrapping = false;
113 
114  /// Returns the largest value of the type for when the result would be
115  /// too large.
116  static constexpr T too_large(const char*)
117  {
118  return std::numeric_limits<T>::max();
119  }
120 
121  /// Returns the smallest value of the type for when the result would be
122  /// too small.
123  static T constexpr too_small(const char*)
124  {
125  return std::numeric_limits<T>::min();
126  }
127 
128  /// Throws an `overflow_div_zero` exception.
129  static T constexpr div_zero(const char* who)
130  {
131  throw overflow_div_zero(who);
132  }
133 };

184 /// Is type `A` wide enough to hold ever value of type `B`?
185 template<class A, class B>
186 constexpr bool is_as_wide_as()
187 {
188  if (std::is_signed<B>::value == std::is_signed<A>::value)
189  return sizeof(B) <= sizeof(A);
190 
191  if (std::is_unsigned<B>::value && std::is_signed<A>::value)
192  return sizeof(B) < sizeof(A);
193 
194  return false;
195 };
196 
197 /// Does type `A` include values lower than `B_MIN`?
198 template<class A, class B>
199 constexpr bool goes_lower_than()
200 {
201  if (std::is_unsigned<A>::value) return false;
202 
203  if (std::is_unsigned<B>::value) return true;
204 
205  return sizeof(A) > sizeof(B);
206 };
207 
208 /// Does type `A` include values higher than `B_MAX`?
209 template<class A, class B>
210 constexpr bool goes_higher_than()
211 {
212  if (sizeof(A) == sizeof(B))
213  return std::is_unsigned<A>::value && std::is_signed<B>::value;
214 
215  return sizeof(A) > sizeof(B);
216 };

218 /// Gets the minimum value of type `T` in type `Repr`.

273 /// Specialized based on the ranges of the `To` and `From` types, and the
274 /// `Policy`.
275 ///
276 /// A structure type `Convert<To, From, Policy>` defines a public static member
277 /// `To convert(From)` that converts from `From` to `To`, acting according to
278 /// `Policy` if the value does not fit. For example, we can perform a
279 /// saturating conversion from `short` to `signed char`:
280 ///
281 /// ```cpp
282 /// short s = 130;
283 /// signed char c = Convert<signed char, short, policy::saturating>::convert(s);
284 /// assert(c == 127);
285 /// ```
286 ///
287 /// Additionally, if all values of type `From` fit in type `To` (that is, if
288 /// `To` is as wide as `From`), then `Convert<To, From, Policy>` defines a
289 /// public static member `To widen(From)`. This can be used to statically
290 /// allow only widening conversions.
291 template <class To,
292  class from,
293  template <class> class Policy = policy::throwing,
294  class Enable = void>
295 struct Convert;
296 
297 /// This specialization implements widening (non-lossy) conversions.
298 template <class To, class From, template <class> class Policy>
299 struct Convert<To, From, Policy,
300  std::enable_if_t<detail::is_as_wide_as<To, From>()>>
301 {
302  /// Widens to `To` from `From`.
303  ///
304  /// Because this is a widening, policy `Policy` is irrelevant.
305  static constexpr To convert(From from)
306  {
307  return static_cast<To>(from);
308  }
309 
310  /// Only for widening conversions do we provide the widen function.
311  static constexpr To widen(From from)
312  {
313  return static_cast<To>(from);
314  }
315 };
316 
317 /// Non-wrapping conversion where the value might be too low.
318 template <class To, class From, template <class> class Policy>
319 struct Convert<To, From, Policy,
320  std::enable_if_t<detail::goes_lower_than<From, To>()
321  && !detail::goes_higher_than<From, To>()
322  && !Policy<To>::is_wrapping>>
323 {
324  /// Converts to `To` from `From`, handling bounds errors according to
325  /// policy `Policy`.
326  static constexpr To convert(From from)
327  {
328  if (detail::is_too_small_for<To>(from))
329  return Policy<To>::too_small("Convert");
330  return static_cast<To>(from);
331  }
332 };

335 template <class To, class From, template <class> class Policy>
336 struct Convert<To, From, Policy,
337  std::enable_if_t<detail::goes_lower_than<From, To>()
338  && detail::goes_higher_than<From, To>()
339  && !Policy<To>::is_wrapping>>

419 /// Non-wrapping, signed integers.
420 template <class T, template<class> class P>
421 class Checked<T, P,
422  std::enable_if_t<std::is_signed<T>::value && !P<T>::is_wrapping>>
423 {
424 private:
425  using unsigned_t = std::make_unsigned_t<T>;
426  using policy_t = P<T>;
427 
428  T value_;
429 
430  static constexpr T T_MIN_ = std::numeric_limits<T>::min();
431  static constexpr T T_MAX_ = std::numeric_limits<T>::max();
432 
433  template <typename U>
434  static constexpr Checked rebuild_(U value)
435  {
436  return static_cast<T>(value);
437  }
438 
439  template <class U, template <class> class Q, class F>
440  friend class Checked;
441 
442 public:
443  /// Non-converting constructor, defaults to 0.
444  constexpr Checked(T value = T()) : value_(value)
445  { }

448  template <class U>
449  constexpr Checked(U value) : value_(Convert<T, U, P>::convert(value))
450  { }

453  template <class U, template<class> class Q>
454  constexpr Checked(Checked<U, Q> other) : value_(Convert<T, U, P>::widen(other.value_))
455  { }

458  constexpr T get() const
459  {
460  return value_;
461  }

464  /// to the old policy `P`.

471  /// Checked negation.
472  constexpr Checked operator-() const
473  {
474  if (std::is_signed<T>::value && value_ == T_MIN_)
475  return policy_t::too_large("Checked::operator-()");
476  else
477  return rebuild_(-value_);
478  }

480  /// Absolute value.
481  constexpr unsigned_t abs() const
482  {
483  if (value_ == T_MIN_) {
484  return unsigned_t(std::numeric_limits<T>::max()) + 1;
485  } else if (value_ < 0) {
486  return unsigned_t(-value_);
487  } else {
488  return unsigned_t(value_);
489  }
490  }

492  /// Checked addition.
493  constexpr Checked operator+(Checked other) const
494  {
495 #if __has_builtin(__builtin_add_overflow)
496  Checked result;
497  if (__builtin_add_overflow(value_, other.value_, &result.value_)) {
498  if (value_ >= 0)
499  return policy_t::too_large("Checked::operator+(Checked)");
500  else
501  return policy_t::too_small("Checked::operator+(Checked)");
502  } else {
503  return result;
504  }
505 #else
506  if (value_ >= 0) {
507  if (other.value_ > T_MAX_ - value_)
508  return policy_t::too_large("Checked::operator+(Checked)");
509  } else {
510  if (other.value_ < T_MIN_ - value_)
511  return policy_t::too_small("Checked::operator+(Checked)");
512  }
513 
514  return rebuild_(value_ + other.value_);
515 #endif
516  }
517 
518  /// Checked subtraction.
519  constexpr Checked operator-(Checked other) const
520  {
521 #if __has_builtin(__builtin_sub_overflow)
522  Checked result;
523  if (__builtin_sub_overflow(value_, other.value_, &result.value_)) {
524  if (value_ >= 0)
525  return policy_t::too_large("Checked::operator-(Checked)");
526  else
527  return policy_t::too_small("Checked::operator-(Checked)");
528  } else {
529  return result;
530  }
531 #else
532  if (value_ >= 0) {
533  if (other.value_ < value_ - T_MAX_)
534  return policy_t::too_large("Checked::operator-(Checked)");
535  } else {
536  if (other.value_ > value_ - T_MIN_)
537  return policy_t::too_small("Checked::operator-(Checked)");
538  }

542  }

545  constexpr Checked operator*(Checked other) const
546  {
547  auto overflow = [=]() {
548  if (detail::same_sign(value_, other.value_))
549  return policy_t::too_large("Checked::operator*(Checked)");
550  else

552  };
553
